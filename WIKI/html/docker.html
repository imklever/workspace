Docker

install:

in centos:

yum install -y docker

in ubuntu:

apt-get install docker.io

command list:

Container:

docker ps -a
docker create  -i -t -p80:80 --name <container name> <image name>   #create container
docker run -d -p80:80 --name <container name> <image name>          #create and run the container with daemon mode
docker run -i -t -d -p80:80 --name <container name> <image name>    #create and run the container with interactive mode
docker rename <container name> <new name>                           #rename a container

docker rm  <container name>                                         #delete container

docker inspect  <container name>                                    #output info of the container

docker start <container name>                                       #start a container
docker stop <container name>                                        #stop a container

docker export -o <container file> <container ID>                    #export a container
docker import <container file>                                      #import a container

docker exec <container name> <command>                              #execute command in docker 
docker attach -i <container name>                                   #go into the container, then you can use it in interactive mode

Image:

docker images
docker commit -m <'message'> <container ID> <REPOSITORY:TAG>    #commit the changes into the base image to create a new image
docker save -o <image file> <image ID>                          #save the image into a image file
docker load -i <image file>                                     #load the image from a image file
docker rmi  <image name>                                        #delete image
docker tag <image ID> <REPOSITORY:TAG>

Data volume:

docker volume create --name test_volume              #create a data volume
docker volume ls                                     #list data volume
docker volume inspect test_volume                    #show the info of data volume
docker volume rm test_volume                         #delete a data volume
docker create -ti -v <path in host>:<path in container>:ro -v <volume name>:<path in container>:ro --name <container name> <image name>      #mount the data volume into the container,ro means read only
    eg: docker create -ti -v /root:/volumes/volume2:ro -v volume1:/volumes/volume1 --name test_container ubuntu
